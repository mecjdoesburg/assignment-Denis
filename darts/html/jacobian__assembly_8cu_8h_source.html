11 // Option 1: assemble in single pass, one block (row) per thread
12 template<typename IndexType, typename ValueType, typename InterpValueType,
13  int _BLOCK_SIZE>
14  __global__ void
15  jacobian_assembly_kernel (const int n_blocks,
16  const ValueType dt, const IndexType *rows_ptr,
17  const IndexType *cols_ind, ValueType *Jac,
18  ValueType *RHS, const ValueType *X,
19  const ValueType *tran, const ValueType *PV,
20  const InterpValueType *acc1,
21  const InterpValueType *acc2,
22  const InterpValueType *flu1,
23  const InterpValueType *flu2)
24  {
25  const unsigned i = blockIdx.x * _BLOCK_SIZE + threadIdx.x; // global thread index
26  if (i > n_blocks - 1)
27  return;
28  //IndexType start = rows_ptr[i];
29  //IndexType end = rows_ptr[i + 1];
30  IndexType jac_diag_idx;
31  ValueType p_diff, gamma_p_diff, jtran_dt;
32 
33  //__shared__ ValueType wkspc[BLOCK_SIZE * 6];
34  //ValueType *jac_d = wkspc + 6 * threadIdx.x;
35  //ValueType *rhs = wkspc + 6 * threadIdx.x + 4;
36  ValueType jac_d[4];
37  ValueType rhs[2];
38 
39  jac_d[0] = 0;
40  jac_d[1] = 0;
41  jac_d[2] = 0;
42  jac_d[3] = 0;
43  rhs[0] = 0;
44  rhs[1] = 0;
45 
46  for (IndexType j = rows_ptr[i]; j < rows_ptr[i + 1]; j++)
47  {
48  unsigned cl = cols_ind[j];
49 
50  if (i == cl)
51  {
52  // fill diagonal part
53  rhs[0] += PV[i] * (acc1[4 * i] - acc1[4 * i + 3]);
54  rhs[1] += PV[i] * (acc2[4 * i] - acc2[4 * i + 3]);
55 
56  jac_diag_idx = j;
57  jac_d[0] += PV[i] * acc1[4 * i + 1];
58  jac_d[1] += PV[i] * acc1[4 * i + 2];
59  jac_d[2] += PV[i] * acc2[4 * i + 1];
60  jac_d[3] += PV[i] * acc2[4 * i + 2];
61  }
62  else
63  {
64  // fill offdiagonal part + contribute to diagonal
65 
66  p_diff = X[2 * cl] - X[2 * i];
67 
68  if (cl < i)
69  jtran_dt = tran[j - i] * dt;
70  else
71  jtran_dt = tran[j - i - 1] * dt;
72  gamma_p_diff = jtran_dt * p_diff;
73 
74  if (p_diff < 0)
75  {
76  //outflow
77  rhs[0] -= gamma_p_diff * flu1[4 * i];
78  rhs[1] -= gamma_p_diff * flu2[4 * i];
79 
80  jac_d[0] -= gamma_p_diff * flu1[4 * i + 1];
81  jac_d[1] -= gamma_p_diff * flu1[4 * i + 2];
82  jac_d[2] -= gamma_p_diff * flu2[4 * i + 1];
83  jac_d[3] -= gamma_p_diff * flu2[4 * i + 2];
84 
85  Jac[4 * j + 0] = -jtran_dt * flu1[4 * i];
86  jac_d[0] -= -jtran_dt * flu1[4 * i];
87 
88  Jac[4 * j + 2] = -jtran_dt * flu2[4 * i];
89  jac_d[2] -= -jtran_dt * flu2[4 * i];
90 
91  Jac[4 * j + 1] = 0;
92  Jac[4 * j + 3] = 0;
93  }
94  else
95  {
96  rhs[0] -= gamma_p_diff * flu1[4 * cl];
97  rhs[1] -= gamma_p_diff * flu2[4 * cl];
98 
99  jac_d[0] += jtran_dt * flu1[4 * cl];
100  jac_d[2] += jtran_dt * flu2[4 * cl];
101 
102  Jac[4 * j + 0] = -gamma_p_diff * flu1[4 * cl + 1] - jtran_dt * flu1[4 * cl];
103  Jac[4 * j + 1] = -gamma_p_diff * flu1[4 * cl + 2];
104  Jac[4 * j + 2] = -gamma_p_diff * flu2[4 * cl + 1] - jtran_dt * flu2[4 * cl];
105  Jac[4 * j + 3] = -gamma_p_diff * flu2[4 * cl + 2];
106  }
107  }
108  }
109 
110  RHS[2 * i + 0] = rhs[0];
111  RHS[2 * i + 1] = rhs[1];
112  Jac[4 * jac_diag_idx + 0] = jac_d[0];
113  Jac[4 * jac_diag_idx + 1] = jac_d[1];
114  Jac[4 * jac_diag_idx + 2] = jac_d[2];
115  Jac[4 * jac_diag_idx + 3] = jac_d[3];
116 
117  }
